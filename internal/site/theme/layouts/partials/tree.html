{{/* File tree rail. Only the current file's own folder is rendered server-side. Emitting the whole tree here costs files × pages — on a 1,371-file repo that was 29%+ of every page and the bulk of the output (TDS-61). The full tree is hydrated on demand by app.js from /index.json, which the command palette already fetches, so browsing costs one shared cached request instead of a copy per page. With JS off the rail still shows where you are and what sits beside you, and the Explorer table remains the complete, paginated inventory. */}} {{- $cur := .RelPermalink -}} {{- $folder := .Params.folder -}}
Files
{{ with $folder }} {{ $siblings := (where (where site.RegularPages "Section" "files") "Params.folder" .).ByParam "path" }} {{ $n := len $siblings }} {{/* Even one folder can be large — Redmine's db/migrate holds 323 files, and rendering all of them into each of its 323 pages costs more than the rest of the rail put together. Show a window around the current file so "you are here" survives, and leave the rest to the hydrated tree. */}} {{ $cap := 25 }} {{ $shown := $siblings }} {{ $hidden := 0 }} {{ if gt $n $cap }} {{ $i := 0 }} {{ range $k, $p := $siblings }}{{ if eq $p.RelPermalink $cur }}{{ $i = $k }}{{ end }}{{ end }} {{ $start := cond (gt $i 12) (sub $i 12) 0 }} {{ if gt (add $start $cap) $n }}{{ $start = sub $n $cap }}{{ end }} {{ $shown = first $cap (after $start $siblings) }} {{ $hidden = sub $n $cap }} {{ end }}
{{ . }}{{ if gt $n $cap }} {{ $n }}{{ end }}
